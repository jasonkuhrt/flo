Usage: gum filter [<options> ...] [flags]

Filter items from a list

Arguments:
  [<options> ...]    Options to filter.

Flags:
  -h, --help                       Show context-sensitive help.
  -v, --version                    Print the version number

      --indicator="•"              Character for selection
                                   ($GUM_FILTER_INDICATOR)
      --selected=,...              Options that should start as
                                   selected (selects all if given *)
                                   ($GUM_FILTER_SELECTED)
      --[no-]show-help             Show help keybinds ($GUM_FILTER_SHOW_HELP)
      --selected-prefix=" ◉ "      Character to indicate selected items (hidden
                                   if limit is 1) ($GUM_FILTER_SELECTED_PREFIX)
      --unselected-prefix=" ○ "
                                   Character to indicate unselected
                                   items (hidden if limit is 1)
                                   ($GUM_FILTER_UNSELECTED_PREFIX)
      --header=""                  Header value ($GUM_FILTER_HEADER)
      --placeholder="Filter..."    Placeholder value ($GUM_FILTER_PLACEHOLDER)
      --prompt="> "                Prompt to display ($GUM_FILTER_PROMPT)
      --width=0                    Input width ($GUM_FILTER_WIDTH)
      --height=0                   Input height ($GUM_FILTER_HEIGHT)
      --value=""                   Initial filter value ($GUM_FILTER_VALUE)
      --reverse                    Display from the bottom of the screen
                                   ($GUM_FILTER_REVERSE)
      --[no-]fuzzy                 Enable fuzzy matching; otherwise match from
                                   start of word ($GUM_FILTER_FUZZY)
      --[no-]fuzzy-sort            Sort fuzzy results by their scores
                                   ($GUM_FILTER_FUZZY_SORT)
      --timeout=0s                 Timeout until filter command aborts
                                   ($GUM_FILTER_TIMEOUT)
      --input-delimiter="\n"       Option delimiter when reading from STDIN
                                   ($GUM_FILTER_INPUT_DELIMITER)
      --output-delimiter="\n"      Option delimiter when writing to STDOUT
                                   ($GUM_FILTER_OUTPUT_DELIMITER)
      --[no-]strip-ansi            Strip ANSI sequences when reading from STDIN
                                   ($GUM_FILTER_STRIP_ANSI)

Selection
  --limit=1          Maximum number of options to pick
  --no-limit         Pick unlimited number of options (ignores limit)
  --select-if-one    Select the given option if there is only one
  --[no-]strict      Only returns if anything matched. Otherwise return Filter
